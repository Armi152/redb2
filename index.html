    // --- Novemberen belüli random dátum generálása ---

    const now = new Date();

    const year = now.getFullYear();

    const month = 10; // November (0 = január, 10 = november)

    

    // Random nap 1–30 között

    const day = Math.floor(Math.random() * 30) + 1;


    const randomDate = new Date(year, month, day);


    const formattedDate = randomDate.toLocaleDateString("hu-HU", {

      year: "numeric",

      month: "2-digit",

      day: "2-digit"

    });


    document.getElementById("date").textContent = formattedDate;
